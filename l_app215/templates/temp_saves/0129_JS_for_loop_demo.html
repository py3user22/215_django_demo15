_____# 3 conditions to work:
1- start count value must be assigned
2- end-count value assigned
3- incrementor to change value of i on each loop

for (var i = 1; i <=3; i++) {
console.log(i)
}
console.log('Go')

______// Define an array of constant data
const data = [10, 20, 30, 40, 50];

______// Use a for loop to iterate over the array
for (let i = 0; i < data.length; i++) {
// Access each element of the array using the index i
console.log(data[i]);
}


______// Define an array of constant data
const data = ["red", "green", "blue", "yellow", "purple"];

______// Use a for...of loop to iterate over the array
for (const element of data) {
______// Access each element of the array directly
console.log(element);
}
